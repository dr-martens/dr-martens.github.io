Alignment of menus

Academic theme contains config\_default\params.toml where users can specify the menu alignment in the “Advanced” section.

# Main menu alignment (l = left, c = center, r = right) and logo options.
main_menu = {align = "l", show_logo = true}

The default is left-aligned (l)

Contact infos

In config/_default/params.toml one can change the contact information in the Contact details section.

Featurette

Publications

How to importing publications from bibtex to a hugo-academic blog?

The solution shown below bases on the this post One nice feature of hugo-academic is that it includes out-of-the-box a “Publications” section, allowing researchers to easily create a list of their publication as a section of the website. Unfortunately, in order to populate that list, users have to manually create one different .md file for each publication, by cutting and pasting several different info (e.g., title, authors, etc.) in a “simple”” template like this one.

# script to parse my publications to single md files for hugo webpage

bibtex_2academic <- function(bibfile,
                             outfold,
                             abstract = FALSE, 
                             overwrite = FALSE,
                             admin.surname=NULL){
  
  require(RefManageR)
  require(dplyr)
  require(stringr)
  require(anytime)
  
  if(is.null(admin.surname)){stop('Surname of webpage admin not specified.')}
  
  # Import the bibtex file and convert to data.frame
  mypubs   <- ReadBib(bibfile, check = "warn", .Encoding = "UTF-8") %>%
    as.data.frame()
  
  print(mypubs)
  
  # assign "categories" to the different types of publications
  mypubs   <- mypubs %>%
    dplyr::mutate(
      pubtype = dplyr::case_when(bibtype == "Article" ~ "2",
                                 bibtype == "Article in Press" ~ "2",
                                 bibtype == "InProceedings" ~ "1",
                                 bibtype == "Proceedings" ~ "1",
                                 bibtype == "Conference" ~ "1",
                                 bibtype == "Conference Paper" ~ "1",
                                 bibtype == "MastersThesis" ~ "3",
                                 bibtype == "PhdThesis" ~ "3",
                                 bibtype == "Manual" ~ "4",
                                 bibtype == "TechReport" ~ "4",
                                 bibtype == "Book" ~ "5",
                                 bibtype == "InCollection" ~ "6",
                                 bibtype == "InBook" ~ "6",
                                 bibtype == "Misc" ~ "0",
                                 TRUE ~ "0"))
  
  # create a function which populates the md template based on the info
  # about a publication
  create_md <- function(x) {
    
    # names of passed arguments
    pass.arg <- names(x)
    
    # define a date and create filename by appending date and start of title
    if("year" %in% pass.arg) {
      x[["date"]] <- paste0(x[["year"]], "-01-01")
    } else {
      x[["date"]] <- "2999-01-01"
    }
    
    filename <- paste(x[["date"]], x[["title"]] %>%
                        str_replace_all(fixed(" "), "_") %>%
                        str_remove_all(fixed(":")) %>%
                        str_sub(1, 20) %>%
                        paste0(".md"), sep = "_")
    # start writing
    if (!file.exists(file.path(outfold, filename)) | overwrite) {
      fileConn <- file.path(outfold, filename)
      print(fileConn)
      write("---", fileConn)
      
      # abstract and optional shortened version.
      if (("abstract" %in% pass.arg) & abstract) {
        write(paste0("abstract: ", x[["abstract"]]), fileConn, append = T)
      } else {
        write("abstract: ", fileConn, append = T)
      }
      write("abstract_short: ", fileConn, append = T)
      
      # Authors
      auth_hugo <- unlist(strsplit(x["author"], " and "))
      
      # replace German Umlaute
      auth_hugo <- str_replace_all(auth_hugo, "ä", "&auml;")
      auth_hugo <- str_replace_all(auth_hugo, "ö", "&ouml;")
      auth_hugo <- str_replace_all(auth_hugo, "ü", "&üuml;")
      auth_hugo <- str_replace_all(auth_hugo, "ß", "&szlig;")
      auth_hugo <- stringi::stri_trans_general(auth_hugo, "latin-ascii")
      write("authors: ", fileConn, append = T)
      for(auth in auth_hugo){
        if(grepl(admin.surname, auth)){ # add admin as author instead of the surname
          write("- admin", fileConn, append = T)
        }else{
          if(grepl(",",auth)){ # check if first and surname are separated by comma
            tmp <- unlist(strsplit(auth, ","))
            write(paste0("- ",trimws(tmp[2])," ",trimws(tmp[1])), fileConn, append = T)
          }else{
            write(paste0("- ",auth), fileConn, append = T)
          }
        }
      }
      
      #date
      write(paste0("date: \"", anydate(x[["date"]]), "\""), fileConn, append = T)
      # doi
      if ("doi" %in% pass.arg){
        write(paste0("doi: \"https://doi.org/",x[["doi"]], "\""), fileConn, append = T)
      }else{
        write("doi: \"\"", fileConn, append = T)
      }

      # Publication type. Legend:
      # 0 = Uncategorized, 1 = Conference paper, 2 = Journal article
      # 3 = Manuscript, 4 = Report, 5 = Book,  6 = Book section
      # Publication details: journal, volume, issue, page numbers and doi link
      if ("journal" %in% pass.arg){ publication <- x[["journal"]] }
      else{ publication <-""}
      if ("volume" %in% pass.arg) publication <- paste(publication, 
                                                       x[["volume"]])
      if ("number" %in% pass.arg) publication <- paste0(publication,
                                                       " (", x[["number"]], ")")
      if ("pages" %in% pass.arg) publication <- paste0(publication,
                                                      ", pp. ", x[["pages"]])
      # if ("doi" %in% pass.arg) publication <- paste0(publication,
      #                                               ", ", paste0("https://doi.org/", 
      #                                                            x[["doi"]]))
      
      write(paste0("publication: \"", publication,"\""), fileConn, append = T)
      write(paste0("publication_short: \"", publication,"\""),fileConn, append = T)
      
      write("publication_types: ", fileConn, append = T)      
      write(paste0("- \"", x[["pubtype"]],"\""), fileConn, append = T)
      
      # title
      if ("title" %in% pass.arg){
        write(paste0("title: \"",x[["title"]],"\""), fileConn, append = T)
      }else{
        write("title: \"\"", fileConn, append = T)
      }
    
      # other possible fields are kept empty. They can be customized later by 
      # editing the created md
      # featured
      write("featured: false", fileConn, append = T)
      # image
      write("image:", fileConn, append = T)
      write("  caption: \"\"", fileConn, append = T)
      write("  focal_point: \"\"", fileConn, append = T)
      write("  preview_only: false", fileConn, append = T)
      # set projects
      write("projects: []", fileConn, append = T)
      write("tags: ", fileConn, append = T)
      write("- \"\"", fileConn, append = T)
      #links
      if ("preprint" %in% pass.arg){
        write(paste0("url_preprint: ",x[["preprint"]]), fileConn, append = T)
      }else{
        write("url_preprint: \"\"", fileConn, append = T)
      }
      write("url_code: \"\"", fileConn, append = T)
      write("url_dataset: \"\"", fileConn, append = T)
      write("url_pdf: \"\"", fileConn, append = T)
      write("url_poster: \"\"", fileConn, append = T)
      write("url_project: \"\"", fileConn, append = T)
      write("url_slides: \"\"", fileConn, append = T)
      write("url_source: \"\"", fileConn, append = T)
      write("url_video : \"\"", fileConn, append = T)
      
      #other stuff
      # write("math = true", fileConn, append = T)
      # write("highlight = true", fileConn, append = T)
      # Featured image
      # write("[header]", fileConn, append = T)
      
      write("---", fileConn, append = T)
    }
  }
  # apply the "create_md" function over the publications list to generate
  # the different "md" files.
  
  apply(mypubs, FUN = function(x) create_md(x), MARGIN = 1)
}

my_bibfile <- here::here("mylib.bib")
out_fold   <- here::here("content/publication")

bibtex_2academic(bibfile  = my_bibfile, 
                 outfold   = out_fold, 
                 abstract  = FALSE,
                 overwrite = TRUE,
                 admin.surname = "YourSurname")

Math

Academic supports a Markdown extension for $\LaTeX$ math. You can enable this feature by toggling the math option in your config/_default/params.toml file.

To render inline or block math, wrap your LaTeX math with $...$ or $$...$$, respectively.

Example math block:

$$\gamma_{n} = \frac{ 
\left | \left (\mathbf x_{n} - \mathbf x_{n-1} \right )^T 
\left [\nabla F (\mathbf x_{n}) - \nabla F (\mathbf x_{n-1}) \right ] \right |}
{\left \|\nabla F(\mathbf{x}_{n}) - \nabla F(\mathbf{x}_{n-1}) \right \|^2}$$

renders as

$$\gamma_{n} = \frac{ \left | \left (\mathbf x_{n} - \mathbf x_{n-1} \right )^T \left [\nabla F (\mathbf x_{n}) - \nabla F (\mathbf x_{n-1}) \right ] \right |}{\left |\nabla F(\mathbf{x}_{n}) - \nabla F(\mathbf{x}_{n-1}) \right |^2}$$

Example inline math $\nabla F(\mathbf{x}_{n})$ renders as $\nabla F(\mathbf{x}_{n})$.

Example multi-line math using the \\ math linebreak:

$$f(k;p_0^*) = \begin{cases} p_0^* & \text{if }k=1, \\
1-p_0^* & \text {if }k=0.\end{cases}$$

renders as

$$f(k;p_0^*) = \begin{cases} p_0^* & \text{if }k=1, \
1-p_0^* & \text {if }k=0.\end{cases}$$

Asides

Academic supports a shortcode for asides, also referred to as notices, hints, or alerts. By wrapping a paragraph in {{% alert note %}} ... {{% /alert %}}, it will render as an aside.

{{% alert note %}}
A Markdown aside is useful for displaying notices, hints, or definitions to your readers.
{{% /alert %}}

renders as

A Markdown aside is useful for displaying notices, hints, or definitions to your readers.
